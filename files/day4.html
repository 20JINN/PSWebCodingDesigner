var 빗변구하는기계 = function (가로길이, 세로길이) {
var a_sqr = 가로길이 * 가로길이 // 제곱
var b_sqr = 세로길이 * 세로길이 // 제곱
var c = Math.sqrt(a_sqr + b_sqr) // 루트 값
return c
}


var 사각형들 = [
{ width: 100, height: 50 },
{ width: 140, height: 90 },
{ width: 170, height: 100 },
{ width: 200, height: 100 }
]


var 빗변들 = [];


for (var 순서 = 0; 순서 < 사각형들.length; 순서++) {
var 빗변 = 빗변구하는기계(사각형들[순서].width, 사각형들[순서].height)

// push는 새로운 값을 Array에 집어 넣습니다.
빗변들.push(빗변)
}
