by.seonyu

☀ 함수(function)란 하나의 로직을 재실행 할 수 있도록 하는 것!
1) function은 비슷한 코드가 반복 될때,
2) 의도적인 기능범위의 구분으로 유지보수 편의상 향상에 목적이 있습니다.
중학교 2학년때 처음 배우는 y=f(x), y= ax+b 일차함수의 정의, 역할과 같습니다! 아래는 function을 정의하는 기본 문법입니다.
var 기능이름 = function() {
// 기능을 호출 했을때 실행할 코드
console.log("function테스트");
console.log("유용하죠?");
console.log("유용해요");
}
참고 : 원래 아래와 같은 방법으로도 사용 가능하지만 변수 선언이라는 일관된 문법을 적용하기 위해서 수업에서 아래 형태는 사용하지 않습니다.
function 기능이름() {
//블라블라
}
☀기능의 실행
기능을 정의하는 것은 다른 유형의 변수를 선언하는 것과 마찬가지로 컴퓨터에게 기능을 알려주기만 한 것이지 실행한 것이 아닙니다. 기능을 실행 하려면 아래와 같이 실행 해줍니다.
기능이름();
<Parameter, 매개변수>

by.seonyu

☀매개변수는 양편에 관계를 맺어주는 변수.

매개변수(Parameter)를 활용하면 훨씬 효율적인 기능을 만들 수가 있습니다.
Argument라고 하기도 합니다. 둘의 차이가 궁금하다면 아래 링크에서 확인 해보세요.
함수의 아규먼트와 파라미터의 차이 기능을 정의할 때 function뒤에 괄호가 있습니다. 이곳에 매개변수(Parameter)를 정의합니다.
// 괄호에 두개의 파라미터를 받을 수 있게 정의 했습니다.
var 매개변수_더하기 = function(매개변수1, 매개변수2) {
// 블럭 안에서 변수처럼 사용할 수 있게 됩니다.
// 매개변수 값이 무엇이 오게 될지 함수는 모릅니다.
console.log(매개변수1 + 매개변수2);
}

매개변수_더하기(1, 3);
// console: 4

매개변수_더하기("안", "지용");
// console: 안지용
새로운 용어 / DOM : Document object model
<참고>
☀ 참고 1. 함수에서 실행된 결과를 반환하는 방법 return
특정 기능을 수행하고 결과 값을 다시 활용할 수 있게 해줍니다. 기능의 재사용성이 높아질 확률이 있습니다.
var 매개변수_두개를_더하는_기능 = function (매개변수1, 매개변수2) {
return 매개변수1 + 매개변수2;
}

// 반환된 값을 새로운 변수에 담아둡니다.
var 리턴받은값 = 매개변수_두개를_더하는_기능(2, 3);

console.log("리턴받은값 : " + 리턴받은값)
☀ 참고 2. function 안에 선언된 변수는 해당 블럭 안에서만 참조할 수 있습니다.
var 테스트 = function() {
var 테스트변수 = "안녕?";
console.log(테스트변수);
// 결과 "안녕?"
}

// function 바깥에서 참조하면...?

console.log(테스트변수);
// 결과 undefined
