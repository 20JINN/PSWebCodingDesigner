☀ jQuery 설치하기

☀ 사용법 기본

jQuery에서는 $표시를 기본적으로 사용한다.
또한 모든 것이 javascript function으로 정의되어 있다.

$("#아이디이름") /* html 상에서 #id로 정의한 것을 지칭할 때 사용한다. */
$(".클래스이름") /* html 상에서 .class로 정의한 것을 지칭할 때 사용한다. */
$("태그이름") /* html 상에서 tag로 정의한 것을 지칭할 때 사용한다. */
예시

@html

<input id="todo-input" type="text" >
<button id="add-button">
<div class="delete-button"> </div>
<div id="container"> <ul></ul></div>

@jQuery

var $todoInput = $("#todo-input");
var $addButton = $("#add-button");
var $deleteButton = $(".delete-button");
var $itemList = $("#container ul")
요소 선택하기 - 2

$(".클래스이름").eq(1)
// 같은 클래스이름을 가진 요소들 중 두 번째 요소를 선택합니다.
// Array처럼 번호를 전달해주면 선택 할 수 있습니다.

$(".클래스이름").children()
// 요소의 자식 요소들을 모두 선택합니다.

$(".클래스이름").first()
// 같은 클래스이름을 가진 요소들 중 첫 번째 요소를 선택합니다.

$(".클래스이름").last()
// 같은 클래스이름을 가진 요소들 중 마지막 요소를 선택합니다.

$(".클래스이름").find(".test")
// 같은 클래스이름을 가진 요소의 자식들 중에 test클래스 이름을 가지고 있는 것을 찾습니다.

<제어하기>
☀show()

$("#아이디이름").show()
// css의 display가 none인 요소를 보이게 합니다.

☀hide()

$("#아이디이름").hide()
// css의 display를 none으로 만듭니다.

☀width()

var 가로길이 = $("#아이디이름").width();
// 요소의 width를 반환합니다.
console.log(가로길이);


$("#아이디이름").width("300px");
// 요소의 width를 300px로 만듭니다.

☀height()

var 세로길이 = $("#아이디이름").height();
// 요소의 height를 반환합니다.
console.log(가로길이);


$("#아이디이름").height("300px");
// 요소의 height를 300px로 만듭니다.

☀css()

$("#아이디이름").css({"background-color": "red"})

$("#아이디이름").css({

<유용한 기능>

$("body").append("<div>HTML추가</div>");
// body태그 내용 뒤에 Parameter로 전달한 HTML을 추가합니다.

$("body").prepend("<div>HTML추가</div>");
// body태그 내용 앞에 Parameter로 전달한 HTML을 추가합니다.

$(".클래스이름").remove();
// 요소를 선택하여 HTML에서 삭제합니다.

$(".클래스이름").addClass(" 새로운클래스");
// 해당 요소를 선택하여 `새로운클래스`라는 클래스를 추가합니다.
// FROM: <div class="클래스이름">...
// TO : <div class="클래스이름 새로운클래스">...

$(".클래스이름") .removeClass("새로운클래스");
// 해당 요소를 선택하여 `새로운클래스`라는 클래스가 있다면 제거합니다.
// FROM: <div class="클래스이름 새로운클래스>...
// TO : <div class="클래스이름">...

$(".클래스이름").hasClass("새로운클래스");
// 해당 요소에 `새로운클래스`가 있는지 확인합니다.
// true/false값을 return 해줍니다.

<이벤트 제어>
☀on()

$(".클래스이름").on("click", function() {
console.log("클릭됨");
});

// 요소에 click이벤트가 on되면 뒤에 오는 기능을 실행합니다.

☀off()

$(".클래스이름").on("click", function() {
console.log("클릭됨");
});

$(".클래스이름").off();
// 해당 요소에 걸린 모든 이벤트 제거

$(".클래스이름").off("click");
// 해당 요소에 걸린 click 이벤트만 제거

<자주 사용 하는 이벤트>

사실 용도에 따라 반드시 필요한 경우도 있겠지만 주료 사용 빈도가 높은 것들입니다.

click - 클릭할 때
keyup - 키보드 입력할 때
mouseenter - 마우스가 해당요소에 진입할 때
mouseleave - 마우스가 해당요소에서 떠날 때
mousemove - 마우스가 해당요소에서 움직일 때
resize - 리사이즈 될 때
scroll - 스크롤 될 때

그 외 엄청나게 많은 이벤트가 있습니다. Standard Events 만 보세요. Event list
