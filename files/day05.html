사용법

jQuery에서는 $표시를 기본적으로 사용한다.

또한 모든 것이 javascript function으로 정의되어 있다.

$("#아이디이름") /* html 상에서 #id로 정의한 것을 지칭할 때 사용한다. */
$(".클래스이름") /* html 상에서 .class로 정의한 것을 지칭할 때 사용한다. */
$("태그이름") /* html 상에서 tag로 정의한 것을 지칭할 때 사용한다. */

예시

html

<input id="todo-input" type="text" >
<button id="add-button">
<div class="delete-button"> </div>
<div id="container"> <ul></ul></div>

jQuery

var $todoInput = $("#todo-input");
var $addButton = $("#add-button");
var $deleteButton = $(".delete-button");
var $itemList = $("#container ul")
